Macroekonómia

Macroekonómia je odvetvie ekonómie, ktoré sa zaoberá štúdiom ekonomických javov na celkovej úrovni, teda na úrovni krajiny alebo regiónu. Analyzuje ekonomické vzťahy a procesy, ktoré ovplyvňujú celkovú produkciu, zamestnanosť, infláciu, hospodársky rast a ďalšie makroekonomické ukazovatele.

import matplotlib.pyplot as plt

@interact
def business_analysis(price_1=slider(30, 60, step_size=0.1, default=49.95),
                      price_2=slider(30, 60, step_size=0.1, default=39.95),
                      sales_1=slider(0, 200, step_size=1, default=80),
                      sales_2=slider(0, 200, step_size=1, default=140),
                      fixed_costs=slider(0, 10000, step_size=100, default=4000),
                      variable_cost=slider(0, 20, step_size=0.1, default=7),
                      publisher_percentage=slider(0, 100, step_size=1, default=50)):

    # Calculate total sales and total revenue
    total_sales = sales_1 + sales_2
    total_revenue = price_1 * sales_1 + price_2 * sales_2

    # Calculate total costs
    total_costs = fixed_costs + variable_cost * total_sales

    # Calculate profit
    profit = (publisher_percentage / 100) * total_revenue - total_costs

    # Plot the results
    labels = ['Total Sales', 'Total Revenue', 'Total Costs', 'Profit']
    values = [total_sales, total_revenue, total_costs, profit]

    fig, ax = plt.subplots()
    ax.bar(labels, values)
    ax.set_ylabel('Values')
    ax.set_title('Business Analysis')

    plt.show()
